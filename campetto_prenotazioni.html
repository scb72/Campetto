// ============================================================ // CAMPETTO - Google Apps Script Backend // Incolla questo codice in: script.google.com → Nuovo progetto // Poi: Distribuisci → Nuova distribuzione → App Web // Esegui come: Me | Accesso: Chiunque // Copia l'URL della Web App e incollalo nel file HTML // ============================================================ const SHEET_NAME_BOOKINGS = "Prenotazioni"; const SHEET_NAME_MEMBERS = "Soci"; const SLOT_HOURS_START = 8; // prima ora prenotabile const SLOT_HOURS_END = 22; // ultima ora (22 = slot 22:00-23:00) // ── Punto di ingresso GET (usato per JSONP / redirect) ─────── function doGet(e) { return handleRequest(e); } // ── Punto di ingresso POST ─────────────────────────────────── function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter || {}; const action = params.action; let result; try { switch (action) { case "getSlots": result = getSlots(params); break; case "book": result = book(params); break; case "cancelBooking": result = cancelBooking(params); break; case "getMemberInfo": result = getMemberInfo(params); break; default: result = { ok: false, error: "Azione sconosciuta" }; } } catch (err) { result = { ok: false, error: err.message }; } const output = ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); return output; } // ── Inizializza il foglio se non esiste ────────────────────── function initSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // Foglio Prenotazioni let bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); if (!bookSheet) { bookSheet = ss.insertSheet(SHEET_NAME_BOOKINGS); bookSheet.appendRow(["ID", "Data", "Ora", "Nome", "CodiceSocio", "Timestamp"]); bookSheet.getRange(1, 1, 1, 6).setFontWeight("bold"); } // Foglio Soci let membSheet = ss.getSheetByName(SHEET_NAME_MEMBERS); if (!membSheet) { membSheet = ss.insertSheet(SHEET_NAME_MEMBERS); membSheet.appendRow(["CodiceSocio", "Nome", "Cognome", "Email", "Attivo"]); membSheet.getRange(1, 1, 1, 5).setFontWeight("bold"); // Aggiungi qualche socio di esempio membSheet.appendRow(["SOCIO001", "Mario", "Rossi", "mario@example.com", true]); membSheet.appendRow(["SOCIO002", "Luigi", "Verdi", "luigi@example.com", true]); membSheet.appendRow(["SOCIO003", "Anna", "Bianchi", "anna@example.com", true]); membSheet.appendRow(["ADMIN001", "Admin", "Campo", "admin@example.com", true]); } } // ── Recupera gli slot di una settimana ─────────────────────── // params: weekStart (YYYY-MM-DD, lunedì della settimana) function getSlots(params) { initSheets(); const weekStart = params.weekStart; // es. "2025-06-02" if (!weekStart) return { ok: false, error: "weekStart mancante" }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); const data = bookSheet.getDataRange().getValues(); // Costruisci array dei 7 giorni della settimana const days = []; const start = new Date(weekStart + "T00:00:00"); for (let d = 0; d < 7; d++) { const day = new Date(start); day.setDate(start.getDate() + d); days.push(formatDate(day)); } // Raccogli prenotazioni per questa settimana const booked = {}; // "YYYY-MM-DD_HH" → { nome, codiceSocio } for (let r = 1; r < data.length; r++) { const row = data[r]; const rowDate = row[1]; // es. "2025-06-02" let rowDate; if (row[1] instanceof Date) { rowDate = formatDate(row[1]); } else { rowDate = String(row[1]).trim(); } } // Costruisci griglia risposta const slots = {}; for (const day of days) { slots[day] = {}; for (let h = SLOT_HOURS_START; h <= SLOT_HOURS_END; h++) { const key = day + "_" + String(h).padStart(2, "0"); slots[day][h] = booked[key] || null; } } return { ok: true, slots }; } // ── Prenota uno slot ───────────────────────────────────────── // params: date, hour, nome, codiceSocio (vuoto se non socio), guestToken function book(params) { initSheets(); const { date, hour, nome, codiceSocio, guestToken } = params; if (!date || hour === undefined || !nome) { return { ok: false, error: "Parametri mancanti" }; } const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); // Verifica slot libero const existing = findBooking(bookSheet, date, hour); if (existing) return { ok: false, error: "Slot già occupato" }; const isMember = codiceSocio && codiceSocio.trim() !== ""; if (isMember) { // Verifica che il codice socio esista if (!memberExists(ss, codiceSocio)) { return { ok: false, error: "Codice socio non valido" }; } // Regola: il socio può prenotare solo se non ha slot futuri non ancora consumati const now = new Date(); const blockingSlot = findFutureUnconsumedSlot(bookSheet, codiceSocio, now); if (blockingSlot) { return { ok: false, error: "Hai già una prenotazione futura (" + blockingSlot.date + " ore " + blockingSlot.hour + ":00). Potrai prenotare di nuovo dopo averla utilizzata." }; } } else { // Non socio: controlla se ha già prenotato (tramite guestToken) if (guestToken) { const guestPrev = findGuestBooking(bookSheet, guestToken); if (guestPrev) { return { ok: false, error: "I non soci possono effettuare solo una prenotazione." }; } } } // Genera ID prenotazione const bookingId = "BK" + Date.now(); bookSheet.appendRow([ bookingId, date, parseInt(hour), nome.trim(), codiceSocio || ("GUEST_" + (guestToken || "???")), new Date().toISOString() ]); return { ok: true, bookingId }; } // ── Cancella una prenotazione (solo se futura e stessa identità) ── function cancelBooking(params) { initSheets(); const { bookingId, codiceSocio, guestToken } = params; if (!bookingId) return { ok: false, error: "ID prenotazione mancante" }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); const data = bookSheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (data[r][0] === bookingId) { // Verifica ownership const rowSocio = String(data[r][4]); const isOwner = (codiceSocio && rowSocio === codiceSocio) || (guestToken && rowSocio === "GUEST_" + guestToken); if (!isOwner) return { ok: false, error: "Non autorizzato" }; // Verifica che sia futura const slotDate = new Date(data[r][1] + "T" + String(data[r][2]).padStart(2,"0") + ":00:00"); if (slotDate <= new Date()) return { ok: false, error: "Non puoi cancellare uno slot passato" }; bookSheet.deleteRow(r + 1); return { ok: true }; } } return { ok: false, error: "Prenotazione non trovata" }; } // ── Info socio ─────────────────────────────────────────────── function getMemberInfo(params) { initSheets(); const { codiceSocio } = params; if (!codiceSocio) return { ok: false, error: "Codice mancante" }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const membSheet = ss.getSheetByName(SHEET_NAME_MEMBERS); const data = membSheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (data[r][0] === codiceSocio && data[r][4]) { return { ok: true, nome: data[r][1], cognome: data[r][2], email: data[r][3] }; } } return { ok: false, error: "Socio non trovato o non attivo" }; } // ── Helper ─────────────────────────────────────────────────── function formatDate(d) { return d.getFullYear() + "-" + String(d.getMonth() + 1).padStart(2, "0") + "-" + String(d.getDate()).padStart(2, "0"); } function findBooking(sheet, date, hour) { const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (data[r][1] === date && parseInt(data[r][2]) === parseInt(hour)) return data[r]; } return null; } function memberExists(ss, codiceSocio) { const sheet = ss.getSheetByName(SHEET_NAME_MEMBERS); const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (data[r][0] === codiceSocio && data[r][4]) return true; } return false; } function findFutureUnconsumedSlot(sheet, codiceSocio, now) { const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (data[r][4] === codiceSocio) { const slotDate = new Date(data[r][1] + "T" + String(data[r][2]).padStart(2,"0") + ":00:00"); if (slotDate > now) { return { date: data[r][1], hour: data[r][2] }; } } } return null; } function findGuestBooking(sheet, guestToken) { const data = sheet.getDataRange().getValues(); const tokenKey = "GUEST_" + guestToken; for (let r = 1; r < data.length; r++) { if (String(data[r][4]) === tokenKey) return data[r]; } return null; }
