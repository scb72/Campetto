const SHEET_NAME_MEMBERS = "Soci"; const SLOT_HOURS_START = 8; const SLOT_HOURS_END = 22; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const params = e.parameter || {}; let result; try { switch (params.action) { case "getSlots": result = getSlots(params); break; case "book": result = book(params); break; case "cancelBooking": result = cancelBooking(params); break; case "getMyNextBooking": result = getMyNextBooking(params); } // ── Inizializza fogli ──────────────────────────────────────── function initSheets() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); if (!bookSheet) { bookSheet = ss.insertSheet(SHEET_NAME_BOOKINGS); membSheet.appendRow(["CodiceSocio", "Nome", "Cognome", "Email", "Attivo"]); membSheet.getRange(1, 1, 1, 5).setFontWeight("bold"); membSheet.appendRow(["SOCIO001", "Mario", "Rossi", "mario@example.com", true]); membSheet.appendRow(["SOCIO002", "Luigi", "Verdi", "luigi@example.com", true]); membSheet.appendRow(["SOCIO003", "Anna", "Bianchi", "anna@example.com", true]); } } // ── Helper: converte qualsiasi valore data in stringa YYYY-MM-DD (ora locale) ── function toDateStr(val) { if (val instanceof Date) { // Usa ora locale dello spreadsheet, non UTC const d = new Date(val.getTime() + (val.getTimezoneOffset() * 60000) * -1); // Oppure più semplice: leggi direttamente i valori locali const local = new Date(val.toLocaleString("en-US", { timeZone: Session.getScriptTimeZone() })); return local.getFullYear() + "-" + String(local.getMonth() + 1).padStart(2, "0") + "-" + String(local.getDate()).padStart(2, "0"); } return String(val).trim(); } // ── getSlots ───────────────────────────────────────────────── function getSlots(params) { initSheets(); const weekStart = params.weekStart; if (!weekStart) return { ok: false, error: "weekStart mancante" }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); const data = bookSheet.getDataRange().getValues(); // Costruisci i 7 giorni della settimana const days = []; const start = new Date(weekStart + "T00:00:00"); for (let d = 0; d < 7; d++) { const day = new Date(start); day.setDate(start.getDate() + d); days.push(toDateStr(day)); } // Raccogli prenotazioni convertendo sempre la data in stringa const booked = {}; for (let r = 1; r < data.length; r++) { const row = data[r]; if (!row[0]) continue; const rowDate = toDateStr(row[1]); if (days.includes(rowDate)) { const key = rowDate + "_" + String(parseInt(row[2])).padStart(2, "0"); booked[key] = { nome: row[3], codiceSocio: String(row[4]), id: row[0] }; } } // Costruisci griglia risposta const slots = {}; for (const day of days) { slots[day] = {}; for (let h = SLOT_HOURS_START; h <= SLOT_HOURS_END; h++) { const key = day + "_" + String(h).padStart(2, "0"); slots[day][h] = booked[key] || null; } } return { ok: true, slots }; } // ── book ───────────────────────────────────────────────────── function book(params) { initSheets(); const { date, hour, nome, codiceSocio, guestToken } = params; if (!date || hour === undefined || !nome) { return { ok: false, error: "Parametri mancanti" }; } const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); if (findBooking(bookSheet, date, hour)) { return { ok: false, error: "Slot già occupato" }; } const isMember = codiceSocio && codiceSocio.trim() !== ""; if (isMember) { if (!memberExists(ss, codiceSocio)) { return { ok: false, error: "Codice socio non valido" }; } const blockingSlot = findFutureUnconsumedSlot(bookSheet, codiceSocio, new Date()); if (blockingSlot) { return { ok: false, error: "Hai già una prenotazione futura (" + blockingSlot.date + " ore " + blockingSlot.hour + ":00). Potrai prenotare di nuovo dopo averla utilizzata." }; } } else { if (guestToken && findGuestBooking(bookSheet, guestToken)) { return { ok: false, error: "I non soci possono effettuare solo una prenotazione." }; } } const bookingId = "BK" + Date.now(); const lastRow = bookSheet.getLastRow() + 1; bookSheet.getRange(lastRow, 1).setValue(bookingId); bookSheet.getRange(lastRow, 2).setValue("'" + date); bookSheet.getRange(lastRow, 3).setValue(parseInt(hour)); bookSheet.getRange(lastRow, 4).setValue(nome.trim()); bookSheet.getRange(lastRow, 5).setValue(codiceSocio || ("GUEST_" + (guestToken || "???"))); bookSheet.getRange(lastRow, 6).setValue(new Date().toISOString()); return { ok: true, bookingId }; } // ── cancelBooking ──────────────────────────────────────────── function cancelBooking(params) { initSheets(); const { bookingId, codiceSocio, guestToken } = params; if (!bookingId) return { ok: false, error: "ID prenotazione mancante" }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const bookSheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); const data = bookSheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (data[r][0] === bookingId) { const rowSocio = String(data[r][4]); const isOwner = (codiceSocio && rowSocio === codiceSocio) || (guestToken && rowSocio === "GUEST_" + guestToken); if (!isOwner) return { ok: false, error: "Non autorizzato" }; const slotDate = new Date(toDateStr(data[r][1]) + "T" + String(parseInt(data[r][2])).padStart(2,"0") + ":00:00"); if (slotDate <= new Date()) return { ok: false, error: "Non puoi cancellare uno slot passato" }; bookSheet.deleteRow(r + 1); return { ok: true }; } } return { ok: false, error: "Prenotazione non trovata" }; } // ── getMyNextBooking ───────────────────────────────────────── function getMyNextBooking(params) { initSheets(); const { codiceSocio, guestToken } = params; const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME_BOOKINGS); const data = sheet.getDataRange().getValues(); const now = new Date(); for (let r = 1; r < data.length; r++) { const row = data[r]; if (!row[0]) continue; const rowCodice = String(row[4]); const match = (codiceSocio && rowCodice.toUpperCase() === codiceSocio.toUpperCase()) || (guestToken && rowCodice === "GUEST_" + guestToken); if (!match) continue; const rowDate = toDateStr(row[1]); const slotDt = new Date(rowDate + "T" + String(parseInt(row[2])).padStart(2,"0") + ":00:00"); if (slotDt > now) { return { ok: true, found: true, date: rowDate, hour: row[2], nome: row[3], id: row[0] }; } } return { ok: true, found: false }; } // ── getMemberInfo ──────────────────────────────────────────── function getMemberInfo(params) { initSheets(); const { codiceSocio } = params; if (!codiceSocio) return { ok: false, error: "Codice mancante" }; const ss = SpreadsheetApp.getActiveSpreadsheet(); const membSheet = ss.getSheetByName(SHEET_NAME_MEMBERS); const data = membSheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][0]).toUpperCase() === codiceSocio.toUpperCase() && data[r][4]) { return { ok: true, nome: data[r][1], cognome: data[r][2], email: data[r][3] }; } } return { ok: false, error: "Socio non trovato o non attivo" }; } // ── Helper ─────────────────────────────────────────────────── function findBooking(sheet, date, hour) { const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (toDateStr(data[r][1]) === date && parseInt(data[r][2]) === parseInt(hour)) return data[r]; } return null; } function memberExists(ss, codiceSocio) { const sheet = ss.getSheetByName(SHEET_NAME_MEMBERS); const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][0]).toUpperCase() === codiceSocio.toUpperCase() && data[r][4]) return true; } return false; } function findFutureUnconsumedSlot(sheet, codiceSocio, now) { const data = sheet.getDataRange().getValues(); for (let r = 1; r < data.length; r++) { if (String(data[r][4]).toUpperCase() === codiceSocio.toUpperCase()) { const slotDate = new Date(toDateStr(data[r][1]) + "T" + String(parseInt(data[r][2])).padStart(2,"0") + ":00:00"); if (slotDate > now) { return { date: toDateStr(data[r][1]), hour: data[r][2] }; } } } return null; } function findGuestBooking(sheet, guestToken) { const data = sheet.getDataRange().getValues(); const tokenKey = "GUEST_" + guestToken; for (let r = 1; r < data.length; r++) { if (String(data[r][4]) === tokenKey) return data[r]; } return null; }
